Voxel Preparation Stage（体素准备阶段）¶

本阶段的作用是 对 LiDAR 点云与伪点云进行统一预处理，并构建 VirConv Block 的标准输入表示。
该阶段不涉及可学习参数，主要由 几何处理与规则化算子 组成，为后续稀疏卷积与融合计算奠定基础。

阶段目标¶

Voxel Preparation Stage 的主要目标包括：

  • 融合真实 LiDAR 点云与伪点云
  • 对点级数据进行坐标对齐与规则化
  • 将不规则点云转换为体素级表示
  • 引入距离分箱等先验信息，增强空间建模能力
  • 为 VirConv Block 构建 STVD 所需的输入结构

本阶段包含的算子¶

本阶段共包含 5 个算子，执行顺序如下：

  1. stvd_input
    构建 STVD 所需的基础输入结构。

  2. point_fusion
    融合 LiDAR 点云与伪点云，形成统一点集。

  3. point_align
    对融合后的点云进行坐标与格式对齐。

  4. voxelization
    将点云转换为稀疏体素表示。

  5. distance_binning
    基于体素中心的空间距离进行分箱编码。

@torch.library.custom_op("virconv::stvd_input", mutates_args=[])
def stvd_input(voxel_coords: torch.Tensor,
               voxel_features: torch.Tensor,
               bin_ids: torch.Tensor,
               discard_rate: float) -> Tuple[torch.Tensor, torch.Tensor]:
    """
    对输入体素执行随机丢弃（Input StVD），用于控制体素数量。

    Args:
        voxel_coords: (M, 3)
            输入体素坐标索引。
        voxel_features: (M, C)
            输入体素特征。
        bin_ids: (M,)
            体素所属的距离 bin（由算子 4 生成）。
        discard_rate: float
            丢弃比例，取值范围 [0, 1)，表示随机丢弃的体素比例。

    Returns:
        voxel_coords_kept: (M', 3)
            丢弃后保留的体素坐标。
        voxel_features_kept: (M', C)
            丢弃后保留的体素特征。
    """
    if voxel_coords.ndim != 2 or voxel_coords.shape[1] != 3:  # 检查坐标
        raise ValueError(f"voxel_coords must be (M,3), got {voxel_coords.shape}")  # 报错信息
    if voxel_features.ndim != 2:  # 检查特征
        raise ValueError(f"voxel_features must be (M,C), got {voxel_features.shape}")  # 报错信息
    if voxel_coords.shape[0] != voxel_features.shape[0]:  # 数量一致性
        raise ValueError("voxel_coords and voxel_features must have same length")  # 报错信息
    if bin_ids.shape[0] != voxel_coords.shape[0]:  # bin 对齐
        raise ValueError("bin_ids must have same length as voxel_coords")  # 报错信息

    if discard_rate <= 0.0:  # 不丢弃
        return voxel_coords.clone(), voxel_features.clone()  # 直接返回
    if discard_rate >= 1.0:  # 全丢弃（极端情况）
        return voxel_coords[:0], voxel_features[:0]  # 返回空

    # 为每个体素生成随机保留标记
    keep_prob = 1.0 - discard_rate  # 保留概率
    rand = torch.rand(
        (voxel_coords.shape[0],),
        device=voxel_coords.device,
        dtype=torch.float32,
    )
    keep_mask = rand < keep_prob  # 是否保留

    voxel_coords_kept = voxel_coords[keep_mask]  # 筛选坐标
    voxel_features_kept = voxel_features[keep_mask]  # 筛选特征

    return voxel_coords_kept, voxel_features_kept

@torch.library.custom_op("virconv::point_fusion", mutates_args=[])
def point_fusion(points_lidar: torch.Tensor, points_virtual: torch.Tensor) -> torch.Tensor:
    """
    将 LiDAR 点云与 Virtual points 进行融合拼接，并返回融合后的点云张量。

    Args:
        points_lidar: (N, C)
            LiDAR 点云张量，每行表示一个点，C 为点特征维度。
        points_virtual: (M, C)
            虚拟点云张量，每行表示一个点，C 为点特征维度。

    Returns:
        points_fused: (N+M, C)
            融合后的点云张量，按行拼接 [points_lidar; points_virtual]。
    """
    if points_lidar.ndim != 2:  # 检查维度
        raise ValueError(f"points_lidar must be 2D (N,C), got {points_lidar.shape}")  # 报错信息
    if points_virtual.ndim != 2:  # 检查维度
        raise ValueError(f"points_virtual must be 2D (M,C), got {points_virtual.shape}")  # 报错信息

    if points_lidar.shape[1] != points_virtual.shape[1]:  # 检查特征维度一致
        raise ValueError(
            f"feature dim mismatch: lidar C={points_lidar.shape[1]} vs virtual C={points_virtual.shape[1]}"
        )  # 报错信息

    if points_virtual.numel() == 0:  # 处理空虚拟点
        return points_lidar  # 直接返回
    if points_lidar.numel() == 0:  # 处理空雷达点
        return points_virtual  # 直接返回

    return torch.cat([points_lidar, points_virtual], dim=0)  # 行拼接融合

@torch.library.custom_op("virconv::voxelization", mutates_args=[])
def voxelization(points: torch.Tensor,
                 voxel_size: torch.Tensor,
                 point_cloud_range: torch.Tensor) -> Tuple[torch.Tensor, torch.Tensor]:
    """
    将点云映射到体素网格中，并生成体素坐标与体素特征。

    Args:
        points: (N, C)
            输入点云张量，前 3 维为 (x, y, z)，其余为点特征。
        voxel_size: (3,)
            体素尺寸 [vx, vy, vz]。
        point_cloud_range: (6,)
            点云范围 [x_min, y_min, z_min, x_max, y_max, z_max]。

    Returns:
        voxel_coords: (M, 3)
            非空体素的整数坐标索引 (ix, iy, iz)。
        voxel_features: (M, C)
            每个体素对应的特征，采用体素内点特征的均值聚合。
    """
    if points.ndim != 2:
        raise ValueError(f"points must be 2D (N,C), got {points.shape}")
    if points.shape[1] < 3:
        raise ValueError("points must contain at least xyz coordinates")

    # 统一 device / dtype
    voxel_size = voxel_size.to(device=points.device, dtype=torch.float32)
    point_cloud_range = point_cloud_range.to(device=points.device, dtype=torch.float32)

    xyz = points[:, :3]
    feats = points

    pc_min = point_cloud_range[:3]
    pc_max = point_cloud_range[3:]

    mask = ((xyz >= pc_min) & (xyz < pc_max)).all(dim=1)
    xyz = xyz[mask]
    feats = feats[mask]

    if xyz.numel() == 0:
        empty_coords = torch.empty((0, 3), device=points.device, dtype=torch.int32)
        empty_feats = torch.empty((0, feats.shape[1]), device=points.device, dtype=points.dtype)
        return empty_coords, empty_feats

    voxel_coords = torch.floor((xyz - pc_min) / voxel_size).to(torch.int32)

    unique_coords, inverse = torch.unique(voxel_coords, dim=0, return_inverse=True)

    voxel_features = torch.zeros(
        (unique_coords.shape[0], feats.shape[1]),
        device=feats.device,
        dtype=feats.dtype,
    )
    voxel_features.index_add_(0, inverse, feats)

    counts = torch.zeros((unique_coords.shape[0], 1), device=feats.device, dtype=feats.dtype)
    ones = torch.ones((feats.shape[0], 1), device=feats.device, dtype=feats.dtype)
    counts.index_add_(0, inverse, ones)

    voxel_features = voxel_features / counts.clamp_min(1.0)

    # custom_op 输出不允许 alias：这里本来就是新张量，但 coords 也 clone 一下更稳
    return unique_coords.clone(), voxel_features.clone()

@torch.library.custom_op("virconv::distance_binning", mutates_args=[])
def distance_binning(voxel_coords: torch.Tensor,
                     voxel_size: torch.Tensor,
                     point_cloud_range: torch.Tensor,
                     bin_edges: torch.Tensor) -> torch.Tensor:
    """
    根据体素在空间中的距离，将体素划分到不同距离 bin 中。

    Args:
        voxel_coords: (M, 3)
            体素坐标索引 (ix, iy, iz)。
        voxel_size: (3,)
            体素尺寸 [vx, vy, vz]。
        point_cloud_range: (6,)
            点云范围 [x_min, y_min, z_min, x_max, y_max, z_max]。
        bin_edges: (K,)
            距离分桶边界（递增）。

    Returns:
        bin_ids: (M,)
            每个体素所属的距离 bin 编号，范围 [0, K]。
    """
    if voxel_coords.ndim != 2 or voxel_coords.shape[1] != 3:
        raise ValueError(f"voxel_coords must be (M,3), got {voxel_coords.shape}")

    # 统一 device / dtype
    device = voxel_coords.device
    voxel_size = voxel_size.to(device=device, dtype=torch.float32)
    point_cloud_range = point_cloud_range.to(device=device, dtype=torch.float32)
    bin_edges = bin_edges.to(device=device, dtype=torch.float32)

    pc_min = point_cloud_range[:3]

    centers = (voxel_coords.to(torch.float32) + 0.5) * voxel_size + pc_min  # (M,3)
    distances = torch.linalg.norm(centers, dim=1)  # (M,)

    bin_ids = torch.bucketize(distances, bin_edges)  # (M,)
    return bin_ids.clone()  # 避免 alias
